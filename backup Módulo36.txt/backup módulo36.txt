Sub MonitorZeco()
    Dim pathXLS As String
    Dim pathXLS_Rep As String
    Dim pathCSV As String
    Dim pathHTML As String
    Dim folderDashboard As String
    Dim tempFolder As String
    Dim pathXLS_Anual As String
    Dim pathXLS_Rep_Anual As String

    ' Carpeta TEMP de Windows para evitar conflictos con sincronizaci n de OneDrive
    tempFolder = Environ("TEMP")
    
    ' Nombres de archivo  nicos para los temporales de SAP
    pathXLS = tempFolder & "\zeco_fase1.xlsx"
    pathXLS_Rep = tempFolder & "\zeco_fase2.xlsx"
    pathXLS_Anual = tempFolder & "\zeco_anual(temp).xlsx"
    pathXLS_Rep_Anual = tempFolder & "\zeco_rep_anual(temp).xlsx"

    ' 1. LIMPIEZA INICIAL: Borra archivos f sicos y cierra restos de sesiones anteriores.
    ' Es el punto donde se eliminan los archivos del disco de forma segura.
    Call LimpiarEntorno(pathXLS, pathXLS_Rep, pathXLS_Anual, pathXLS_Rep_Anual)

    ' Ruta del Dashboard en OneDrive
    pathHTML = "T:\Produccion\DOI\PRODUCCION\Segmento PROTOTIPOS\IFA\Aplicaciones\Dashboards\ZECO MONITOR\Dashboard Zeco.html"
    
    folderDashboard = Left(pathHTML, InStrRev(pathHTML, "\"))
   pathCSV = "C:\Users\" & Environ$("USERNAME") & "\TEMP\" & "ZECO " & Format(Date, "dd-mm-yyyy") & ".csv"

    ' 2. Ejecutar proceso de extracci n en SAP
    Call ExtraerDatosSAP(pathXLS, pathXLS_Rep, pathCSV)

    ' 3. Extraer datos anuales SAP y agregar seccion ---ANUAL--- al CSV
    Call ExtraerDatosAnuales(pathXLS_Anual, pathXLS_Rep_Anual, pathCSV)

    ' 4. ABRIR DASHBOARD: Lanzamos el navegador primero.
    If Dir(pathHTML) <> "" Then
        ThisWorkbook.FollowHyperlink pathHTML
    Else
        MsgBox "No se encuentra el archivo Dashboard Zeco.html en la ruta especificada:" & vbCrLf & pathHTML, vbCritical, "Error de Ruta"
        Exit Sub
    End If

    ' 4. ESPERA ACTIVA: Pausamos 10 segundos.
    ' Esto da tiempo a que el comando autom tico de SAP abra los archivos en Excel.
    application.Wait (Now + TimeValue("0:00:10"))
    
    ' 5. CIERRE DEFINITIVO: Ahora que SAP ya los ha abierto, los cerramos por fuerza.
    Call CerrarLibrosSolo(pathXLS, pathXLS_Rep, pathXLS_Anual, pathXLS_Rep_Anual)
End Sub

Private Sub LimpiarYConvertirCSV(rutaOrigen As String, rutaDestino As String)
    Dim wb As Workbook, ws As Worksheet
    Dim lastRow As Long, r As Long, c As Long
    Dim fnum As Integer, lineText As String
    Dim fileNameOnly As String
    Dim resp As VbMsgBoxResult
    Dim colsToExport As Variant
    
    colsToExport = Array(1, 5, 6, 7)
    fileNameOnly = Mid(rutaOrigen, InStrRev(rutaOrigen, "\") + 1)
    application.DisplayAlerts = False
    
    On Error Resume Next
    Set wb = Workbooks(fileNameOnly)
    If wb Is Nothing Then Set wb = Workbooks.Open(rutaOrigen)
    On Error GoTo 0

    If wb Is Nothing Then
        application.DisplayAlerts = True
        MsgBox "No se pudo abrir el archivo de resultados.", vbCritical
        Exit Sub
    End If

    Set ws = wb.Sheets(1)
    ws.Columns("A:A").TextToColumns Destination:=ws.Range("A1"), DataType:=xlDelimited, Comma:=True
    
    lastRow = ws.Cells(ws.Rows.Count, "A").End(xlUp).Row
    ws.Range("A1:G" & lastRow).RemoveDuplicates Columns:=1, header:=xlYes
    lastRow = ws.Cells(ws.Rows.Count, "A").End(xlUp).Row

REINTENTO_ESCRITURA:
    fnum = FreeFile
    On Error Resume Next
    Open rutaDestino For Output As #fnum
    
    If Err.Number <> 0 Then
        If Err.Number = 70 Then
            Err.Clear
            On Error GoTo 0
            resp = MsgBox("El archivo CSV est  bloqueado por el Dashboard o OneDrive." & vbCrLf & _
                          "Cierra el navegador y pulsa 'Reintentar'.", _
                          vbRetryCancel + vbExclamation, "Archivo Bloqueado")
            If resp = vbRetry Then GoTo REINTENTO_ESCRITURA Else Exit Sub
        Else
            MsgBox "Error al escribir CSV: " & Err.Description, vbCritical
            wb.Close SaveChanges:=False
            Exit Sub
        End If
    End If
    On Error GoTo 0
    
    Print #fnum, "N  ECT;Fecha;Proyecto;Descripci n"
    
    For r = 2 To lastRow
        lineText = ""
        For c = 0 To 3
            Dim val As String
            val = Replace(CStr(ws.Cells(r, colsToExport(c)).value), ";", ",")
            lineText = lineText & val & IIf(c < 3, ";", "")
        Next c
        Print #fnum, lineText
    Next r
    
    Close #fnum
    
    wb.Close SaveChanges:=False
    application.DisplayAlerts = True
End Sub

Private Function LeerDatosDesdeExcel(rutaFile As String) As String
    Dim wb As Workbook, ws As Worksheet
    Dim lastRow As Long, i As Long, concatenado As String
    Dim fileNameOnly As String

    fileNameOnly = Mid(rutaFile, InStrRev(rutaFile, "\") + 1)
    application.DisplayAlerts = False

    On Error Resume Next
    Set wb = Workbooks(fileNameOnly)
    If wb Is Nothing Then Set wb = Workbooks.Open(rutaFile, ReadOnly:=True)
    On Error GoTo 0

    If wb Is Nothing Then
        application.DisplayAlerts = True
        Exit Function
    End If

    Set ws = wb.Sheets(1)
    ws.Columns("A:A").TextToColumns Destination:=ws.Range("A1"), DataType:=xlDelimited, Comma:=True, Tab:=True
    lastRow = ws.Cells(ws.Rows.Count, "A").End(xlUp).Row

    For i = 2 To lastRow
        If ws.Cells(i, 1).value <> "" Then
            concatenado = concatenado & ws.Cells(i, 1).value & vbCrLf
        End If
    Next i

    wb.Close SaveChanges:=False
    application.DisplayAlerts = True
    LeerDatosDesdeExcel = concatenado
End Function

Private Function EsperarArchivo(ruta As String, segundos As Integer) As Boolean
    Dim inicio As Single: inicio = Timer
    Do While Timer < inicio + segundos
        If Dir(ruta) <> "" Then
            On Error Resume Next
            Dim f As Integer: f = FreeFile
            Open ruta For Append Access Write As #f
            If Err.Number = 0 Then
                Close #f: EsperarArchivo = True: Exit Function
            End If
            Err.Clear: On Error GoTo 0
        End If
        DoEvents
    Loop
    EsperarArchivo = False
End Function

Private Sub LimpiarEntorno(f1 As String, f2 As String, Optional f3 As String = "", Optional f4 As String = "")
    Dim n1 As String, n2 As String, n3 As String, n4 As String
    Dim wb As Workbook
    n1 = Mid(f1, InStrRev(f1, "\") + 1)
    n2 = Mid(f2, InStrRev(f2, "\") + 1)
    If f3 <> "" Then n3 = Mid(f3, InStrRev(f3, "\") + 1)
    If f4 <> "" Then n4 = Mid(f4, InStrRev(f4, "\") + 1)
    
    On Error Resume Next
    ' Cierre de libros abiertos iterando la colecci n para asegurar el desbloqueo
    For Each wb In Workbooks
        If wb.Name = n1 Or wb.Name = n2 Then wb.Close SaveChanges:=False
        If f3 <> "" And wb.Name = n3 Then wb.Close SaveChanges:=False
        If f4 <> "" And wb.Name = n4 Then wb.Close SaveChanges:=False
    Next wb
    
    DoEvents
    
    ' Eliminaci n f sica de archivos en la carpeta TEMP
    If Dir(f1) <> "" Then Kill f1
    If Dir(f2) <> "" Then Kill f2
    If f3 <> "" And Dir(f3) <> "" Then Kill f3
    If f4 <> "" And Dir(f4) <> "" Then Kill f4
    On Error GoTo 0
End Sub

Private Sub CerrarLibrosSolo(f1 As String, f2 As String, Optional f3 As String = "", Optional f4 As String = "")
    Dim n1 As String, n2 As String, n3 As String, n4 As String
    Dim wb As Workbook
    n1 = Mid(f1, InStrRev(f1, "\") + 1)
    n2 = Mid(f2, InStrRev(f2, "\") + 1)
    If f3 <> "" Then n3 = Mid(f3, InStrRev(f3, "\") + 1)
    If f4 <> "" Then n4 = Mid(f4, InStrRev(f4, "\") + 1)
    
    On Error Resume Next
    application.DisplayAlerts = False
    ' Recorremos la colecci n de libros actuales para cerrar los que SAP acaba de abrir
    For Each wb In Workbooks
        If wb.Name = n1 Or wb.Name = n2 Then wb.Close SaveChanges:=False
        If f3 <> "" And wb.Name = n3 Then wb.Close SaveChanges:=False
        If f4 <> "" And wb.Name = n4 Then wb.Close SaveChanges:=False
    Next wb
    application.DisplayAlerts = True
    DoEvents
    On Error GoTo 0
End Sub

REINTENTO_ANUAL:
    fnum = FreeFile
